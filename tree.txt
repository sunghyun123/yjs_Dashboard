# app/db/db_manager.py
import sqlite3
import logging
from datetime import datetime
from contextlib import closing
from typing import List, Dict, Any, Optional

# 로거 설정
logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)


class DBManager:
    def __init__(self, db_path: str = "schedule.db"):
        """
        초기화 시점에 데이터베이스 경로를 설정하고 테이블 생성 로직을 실행합니다.
        """
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        """DB 테이블에 details와 tags 컬럼이 없으면 추가하거나 생성합니다."""
        with closing(sqlite3.connect(self.db_path)) as conn:
            with conn:
                # 1. 기본 테이블 생성 (details, tags 컬럼 추가)
                conn.execute("""
                             CREATE TABLE IF NOT EXISTS field_schedules
                             (
                                 id INTEGER PRIMARY KEY AUTOINCREMENT,
                                 date TEXT,
                                 location TEXT,
                                 task TEXT,
                                 person TEXT,
                                 details TEXT,
                                 tags TEXT,
                                 category TEXT,
                                 created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
                             )
                             """)

                # 2. [Migration] 기존 DB를 쓰는 경우를 대비해 컬럼이 없으면 추가
                cursor = conn.execute("PRAGMA table_info(field_schedules)")
                columns = [row[1] for row in cursor.fetchall()]
                if "details" not in columns:
                    conn.execute("ALTER TABLE field_schedules ADD COLUMN details TEXT")
                if "tags" not in columns:
                    conn.execute("ALTER TABLE field_schedules ADD COLUMN tags TEXT")

    def upsert_schedule(self, data: Dict[str, Any]) -> str:
        """
        똑같은 날짜+장소 데이터가 있으면 수정(UPDATE), 없으면 새로 저장(INSERT) (Upsert)
        생성(create) 및 수정(update) 명령에서 공통으로 사용됩니다.
        """
        db_data = data.copy()
        db_data['person'] = data.get('person', '-')
        # tags가 리스트 형태면 쉼표로 연결된 문자열로 변환
        if isinstance(db_data.get('tags'), list):
            db_data['tags'] = ",".join(db_data.get('tags', []))
        elif db_data.get('tags') is None:
            db_data['tags'] = ""

        with closing(sqlite3.connect(self.db_path)) as conn:
            # 1. 같은 날짜와 장소가 이미 있는지 확인
            check_sql = "SELECT id FROM field_schedules WHERE date = ? AND location = ?"
            existing = conn.execute(check_sql, (data['date'], data['location'])).fetchone()

            if existing:
                # 2. 존재하면 UPDATE (수정)
                update_sql = """
                             UPDATE field_schedules
                             SET task       = :task,
                                 person     = :person,
                                 details    = :details,
                                 tags       = :tags,
                                 category   = :category,
                                 created_at = CURRENT_TIMESTAMP
                             WHERE id = ?
                             """
                conn.execute(update_sql, {**db_data, "id": existing[0]})
                conn.commit()
                return f"📍 {data['location']} ({data['date']}) 일정이 수정되었습니다. (ID: {existing[0]})"
            else:
                # 3. 존재하지 않으면 INSERT (신규)
                insert_sql = """
                             INSERT INTO field_schedules (date, location, task, person, details, tags, category)
                             VALUES (:date, :location, :task, :person, :details, :tags, :category)
                             """
                cursor = conn.execute(insert_sql, db_data)
                conn.commit()
                return f"➕ {data['location']} ({data['date']}) 신규 일정이 등록되었습니다. (ID: {cursor.lastrowid})"

    def delete_schedule_by_data(self, date: str, location: str) -> Optional[int]:
        """
        날짜와 장소를 기준으로 일정을 찾아 삭제합니다.
        삭제된 일정의 ID를 반환하며, 데이터가 없으면 None을 반환합니다.
        """
        if not date or not location:
            return None

        with closing(sqlite3.connect(self.db_path)) as conn:
            with conn:
                # 1. 대상 데이터 ID 조회
                select_sql = "SELECT id FROM field_schedules WHERE date = ? AND location = ?"
                existing = conn.execute(select_sql, (date, location)).fetchone()

                if existing:
                    target_id = existing[0]
                    # 2. 데이터 삭제
                    delete_sql = "DELETE FROM field_schedules WHERE id = ?"
                    conn.execute(delete_sql, (target_id,))
                    logger.info(f"일정 삭제 완료: ID {target_id} ({date}, {location})")
                    return target_id
                else:
                    return None

    def get_all_schedules_desc(self, target_date: str = None) -> List[Dict[str, Any]]:
        """
        모든 일정 데이터를 최근 날짜 및 등록 순으로 50개까지 가져옵니다. (날짜 필터 없음)
        'dashboard.html'에서 화면을 그릴 때 사용합니다.
        """
        # (사용자가 수정한 사항 반영: 날짜 필터를 없애고 LIMIT 50)
        select_query = """
            SELECT id, date, location, task, person, details, tags, category,
                   datetime(created_at, 'localtime') as created_at
            FROM field_schedules
            ORDER BY date DESC, created_at DESC
            LIMIT 50
        """
        try:
            with closing(sqlite3.connect(self.db_path)) as conn:
                conn.row_factory = sqlite3.Row
                cursor = conn.execute(select_query)
                rows = cursor.fetchall()
                return [dict(row) for row in rows]
        except sqlite3.Error as e:
            logger.error(f"[DB 에러] {e}")
            return []

    def insert_worklog(self, data: Dict[str, Any]) -> int:
        # (이 메서드는 그대로 유지, 단 테이블 생성 로직을 _init_db에 추가해야 함)
        insert_query = """
                       INSERT INTO work_logs (project_name, project_code, regular_workers, daily_workers)
                       VALUES (:project_name, :project_code, :regular_workers, :daily_workers)
                       """
        # List 형태인 작업자 명단을 쉼표로 연결된 문자열로 변환
        db_data = {
            "project_name": data.get("project_name", ""),
            "project_code": data.get("project_code", ""),
            "regular_workers": ", ".join(data.get("regular_workers", [])),
            "daily_workers": ", ".join(data.get("daily_workers", []))
        }

        try:
            with closing(sqlite3.connect(self.db_path)) as conn:
                with conn:
                    cursor = conn.execute(insert_query, db_data)
                    return cursor.lastrowid
        except sqlite3.Error as e:
            logger.error(f"[DB 에러] 작업일지 Insert 중 오류 발생: {e}")
            raise

# app/services/ai_service.py
import json
import logging
from datetime import datetime
from typing import Optional, Dict, Any, List, Literal
from pydantic import BaseModel, Field
from google import genai
from google.genai import types

logger = logging.getLogger(__name__)

# 1. 일정 데이터 구조 (create, update 시 사용)
class ScheduleSchema(BaseModel):
    date: str = Field(description="YYYY-MM-DD")
    location: str = Field(description="현장 위치")
    task: str = Field(description="주요 작업 명칭")
    person: str = Field(default="-", description="담당자 또는 수신자")
    details: str = Field(default="", description="참여자, 장비 등 상세 내용")
    tags: List[str] = Field(default_factory=list, description="야간, 돌발, 교육, 대기 등 키워드")
    category: str = Field(description="공사일정, 이슈보고, 일반메모 중 선택")

# 2. [추가] 조회 및 삭제용 파라미터 구조를 명확히 정의합니다!
class QueryParamsSchema(BaseModel):
    date: Optional[str] = Field(default=None, description="필터링할 날짜 (YYYY-MM-DD)")
    location: Optional[str] = Field(default=None, description="필터링할 현장 위치")

# 3. AI 분석 결과 구조 (Dict 대신 위에서 만든 QueryParamsSchema 사용)
class ActionSchema(BaseModel):
    action: Literal["create", "read", "update", "delete", "unknown"] = Field(description="사용자가 요청한 동작 유형")
    query_params: Optional[QueryParamsSchema] = Field(default=None, description="조회(read) 또는 삭제(delete) 시 필요한 필터")
    schedule_data: Optional[ScheduleSchema] = Field(default=None, description="생성(create) 또는 수정(update) 시 최종적으로 저장될 일정 데이터")

class GeminiService:
    def __init__(self, api_key: str):
        self.client = genai.Client(api_key=api_key)
        self.model_name = "gemini-3-flash-preview"

    async def process_command(self, text: str) -> Optional[Dict[str, Any]]:
        current_date = datetime.now().strftime("%Y-%m-%d")

        system_instruction = f"""
        당신은 건설 현장 상황판 관리자입니다. 기준 날짜: {current_date}
        사용자가 입력한 텍스트를 분석하여, 다음 동작 중 하나를 선택하고 그에 필요한 데이터를 JSON으로 만드세요.

        1. create (추가): 새로운 일정을 등록할 때. schedule_data를 채우세요.
        2. read (조회): 특정 날짜나 현장 일정을 보여달라고 할 때. query_params에 필터를 채우세요.
        3. update (수정): 기존 일정을 바꿀 때. schedule_data에 변경된 내용을 담으세요.
        4. delete (삭제): 일정을 지워달라고 할 때. query_params에 대상 필터(date, location)를 정확히 채우세요.
        """

        try:
            response = await self.client.aio.models.generate_content(
                model=self.model_name,
                contents=text,
                config=types.GenerateContentConfig(
                    system_instruction=system_instruction,
                    response_mime_type="application/json",
                    response_schema=ActionSchema,
                    temperature=0.0
                )
            )
            # [중요] dict 대신 명확한 모델을 썼으므로 처리 결과는 여전히 dict로 리턴되게 합니다.
            return json.loads(response.text)
        except Exception as e:
            logger.error(f"상황판 명령 분석 오류: {e}")
            return None

class VisionService:
    def __init__(self, api_key: str):
        self.client = genai.Client(api_key=api_key)
        self.model_name = "gemini-3-flash-preview"

    async def analyze_document(self, image_bytes: bytes, mime_type: str) -> Optional[Dict[str, Any]]:
        prompt = "이 이미지를 분석하여 문서 종류를 분류하고, ERP 입력에 필요한 숫자 데이터를 추출해 JSON으로 응답하세요."

                config=types.GenerateContentConfig(

                    response_schema=AnalysisResult,
                    temperature=0.0
                )
            )
            return json.loads(response.text)
        except Exception as e:

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from typing import Optional
import logging

from app.services.ai_service import GeminiService

from app.core.config import settings

logger = logging.getLogger(__name__)

# 1. 라우터 생성
router = APIRouter(prefix="/api/schedules", tags=["Schedules"])

# 2. 의존성 객체 초기화
ai_svc = GeminiService(api_key=settings.GEMINI_API_KEY)
db = DBManager(db_path="schedule.db")

# 3. Request 모델
class ReportRequest(BaseModel):
    text: str
    class Config:
        json_schema_extra = {
            "example": {"text": "오늘 안양 현장 접지 끝남. 김대리님한테 전달 좀"}
        }

# 4. 엔드포인트 통합 (보고 및 CRUD 명령 처리)
@router.post("", summary="현장 보고 및 스마트 일정 관리 (CRUD)")
async def create_or_update_schedule_by_command(request: ReportRequest):
    """
    사용자의 자연어 명령을 AI가 분석하여, 일정 추가/조회/수정/삭제를 자동으로 수행합니다.
    """
    logger.info(f"새로운 스마트 명령 수신: {request.text}")
    # AI 명령 처리기 호출
    action_data = await ai_svc.process_command(request.text)

    if not action_data:
        raise HTTPException(status_code=400, detail="AI가 명령을 분석하지 못했습니다.")

    action = action_data['action']
    logger.info(f"AI가 판단한 동작: {action}")

    try:
        # 4-1. 추가(create) 및 수정(update) 동작
        if action in ["create", "update"]:
            schedule_data = action_data.get('schedule_data')
            if not schedule_data:
                raise HTTPException(status_code=400, detail="일정 데이터가 누락되었습니다.")

            # insert_schedule 호출 버그 수정 -> upsert_schedule 호출!
            result_msg = db.upsert_schedule(schedule_data)
            return {"message": result_msg, "action": action, "data": schedule_data}

        # 4-2. 조회(read) 동작
        elif action == "read":
            query_params = action_data.get('query_params', {})
            date = query_params.get('date')
            location = query_params.get('location')

            logger.info(f"일정 조회 요청: 날짜={date}, 위치={location}")
            # 대시보드 리다이렉트는 프론트엔드에서 처리하도록 유도
            return {"message": "일정 조회를 시도합니다.", "action": "read", "query_params": query_params}

        # 4-3. 삭제(delete) 동작
        elif action == "delete":
            query_params = action_data.get('query_params', {})
            date = query_params.get('date')
            location = query_params.get('location')

            logger.info(f"일정 삭제 요청: 날짜={date}, 위치={location}")
            if not date or not location:
                raise HTTPException(status_code=400, detail="삭제할 날짜와 위치를 정확히 알 수 없습니다.")

            deleted_id = db.delete_schedule_by_data(date, location)

            if deleted_id:
                return {"message": f"🗑️ {location} ({date}) 일정이 삭제되었습니다. (ID: {deleted_id})", "action": "delete"}
            else:
                return {"message": f"⚠️ {location} ({date}) 데이터를 찾을 수 없어 삭제하지 못했습니다.", "action": "delete"}

        # 4-4. 알 수 없는 동작
        else:
            return {"message": "알 수 없는 명령입니다.", "action": "unknown"}

    except Exception as e:
        logger.error(f"DB 작업 실패 ({action}): {e}")
        raise HTTPException(status_code=500, detail=str(e))

        // 일정 데이터 불러오기 함수
        async function loadSchedules() {
            const board = document.getElementById('scheduleBoard');
            try {
                // 백엔드 API 호출 (오늘을 포함한 미래의 모든 일정 가져옴)
                const response = await fetch('/api/schedules/today');
                const data = await response.json();

                if (data.data.length === 0) {
                    board.innerHTML = '<div class="text-center text-muted py-4">예정된 일정이 없습니다.</div>';
                    return;
                }

                // 1. 데이터를 '날짜별'로 묶어주기 (Grouping)
                const groupedData = data.data.reduce((acc, item) => {
                    const dateKey = item.date || '날짜 미상';
                    if (!acc[dateKey]) acc[dateKey] = [];
                    acc[dateKey].push(item);
                    return acc;
                }, {});

                // 2. 날짜 오름차순으로 정렬 (오늘 -> 내일 -> 모레 순서)
                const sortedDates = Object.keys(groupedData).sort();

                // 3. 화면에 날짜별 구역(Section)을 만들어서 그리기
                let boardHTML = '';

                sortedDates.forEach(date => {
                    // 날짜 헤더 (ex: 📅 2026-03-26 (목))
                    const dayName = getDayName(date);
                    let headerClass = "text-secondary";
                    let dateLabel = date;

                    // '오늘' 날짜면 파란색으로 강조
                    const todayStr = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}-${String(today.getDate()).padStart(2, '0')}`;
                    if (date === todayStr) {
                        headerClass = "text-primary fw-bold";
                        dateLabel = `${date} [오늘]`;
                    }

                    boardHTML += `<h6 class="mt-4 mb-2 pb-1 border-bottom ${headerClass}">📅 ${dateLabel} ${dayName}</h6>`;

                    // 해당 날짜의 일정 카드들 붙이기
                    groupedData[date].forEach(item => {
                        let colorClass = '';
                        if(item.category && item.category.includes('완료')) colorClass = 'category-완료';
                        if(item.category && item.category.includes('이슈')) colorClass = 'category-이슈';

                        // 안전하게 태그 분리 (문자열인 경우 대비)
                        let tagList = [];
                        if (item.tags) {
                            tagList = typeof item.tags === 'string' ? item.tags.split(',') : item.tags;
                        }

        // 1. 스마트 텍스트 명령 전송 함수 (CRUD)
        async function submitTextReport() {
            const textInput = document.getElementById('scheduleText').value;
            const resultDiv = document.getElementById('textResult');
            const loading = document.getElementById('textLoading');

            if (!textInput.trim()) {
                alert('내용을 입력해주세요!');
                return;
            }

            // UI 업데이트 (로딩 켜기)
            loading.style.display = 'inline-block';
            resultDiv.innerHTML = '';

            try {
                const response = await fetch('/api/schedules', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ text: textInput })
                });

                const data = await response.json();

                if (response.ok) {
                    // AI 분석 결과 동작(action)에 따라 분기 처리
                    const action = data.action;
                    resultDiv.innerHTML = `<span class="text-success fw-bold">✅ ${data.message}</span>`;

                    // 1-1. 조회(read) 명령이었으면 대시보드로 이동
                    if (action === "read") {
                        const date = data.query_params?.date;
                        const location = data.query_params?.location;
                        resultDiv.innerHTML = `<span class="text-primary fw-bold">📅 ${location ? location+' ' : ''}${date ? date+' ' : ''}일정을 조회합니다. 곧 이동합니다...</span>`;
                        setTimeout(() => {
                            // 대시보드로 이동 (날짜 파라미터가 있으면 붙여서 이동 - dashboard.html에서 구현 필요)
                            window.location.href = `/dashboard.html${date ? '?date=' + date : ''}`;
                        }, 1000);
                    }
                    // 1-2. 생성/수정/삭제 명령이었으면 입력창 초기화
                    else if (["create", "update", "delete"].includes(action)) {
                        document.getElementById('scheduleText').value = ''; // 입력창 초기화
                    }

                } else {
                    resultDiv.innerHTML = `<div class="alert alert-danger mt-2">❌ 오류: ${data.detail || '알 수 없는 에러'}</div>`;
                }
            } catch (error) {
                resultDiv.innerHTML = `<div class="alert alert-danger mt-2">❌ 서버 통신 오류가 발생했습니다.</div>`;
            } finally {
                loading.style.display = 'none'; // 로딩 끄기
            }
        }
